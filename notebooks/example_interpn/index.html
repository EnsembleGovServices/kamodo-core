import numpy as np
import scipy.interpolate 

x = np.arange(10) 
y = np.arange(10) 
z = np.arange(10)
v = np.ones((10,10,10))
sample_at = 8*np.random.random((30)).reshape(10, 3) 

for i in range(len(x)):
    v[i] = i
