from kamodo import Kamodo, kamodofy

@kamodofy(units='kg')
def mass(x):
    return x**2

@kamodofy(units='m^3')
def vol(y):
    return y**2

k = Kamodo(mass=mass, vol = vol)
k['rho'] = 'mass/vol'
k
